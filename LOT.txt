<script>
const subCountDiv = document.getElementById('subCount');

// Smooth counter animation
async function fetchSubscribers() {
  try {
    const response = await fetch('https://vercel-c4ld.vercel.app/api/subs');
    const data = await response.json();
    const target = Number(data.subscribers);

    if (isNaN(target)) {
      subCountDiv.innerText = 'Unknown';
      return;
    }

    // Animate count
    let count = 0;
    const step = Math.ceil(target / 100); // adjust speed
    const interval = setInterval(() => {
      count += step;
      if (count >= target) {
        count = target;
        clearInterval(interval);
      }
      subCountDiv.innerText = `${count.toLocaleString()} Subscribers`;
    }, 20); // adjust speed
  } catch (error) {
    console.error(error);
    subCountDiv.innerText = 'Unknown';
  }
}

// Start fetching
fetchSubscribers();
</script>
